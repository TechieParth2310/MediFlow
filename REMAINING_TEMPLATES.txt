<script>
const availableDates = {{ available_dates | tojson }};
const dateSelect = document.getElementById('date-select');
const timeSelect = document.querySelector('select[name="appointment_time"]');

dateSelect.addEventListener('change', function() {
    const selectedDate = this.value;
    const dateInfo = availableDates.find(d => d.date === selectedDate);
    
    if (dateInfo && dateInfo.slots) {
        timeSelect.innerHTML = '<option value="">Select time</option>';
        dateInfo.slots.forEach(slot => {
            const option = document.createElement('option');
            option.value = slot.start_time;
            option.textContent = `${slot.start_time} - ${slot.end_time}`;
            timeSelect.appendChild(option);
        });
    }
});
</script>
